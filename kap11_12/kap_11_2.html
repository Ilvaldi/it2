Å løkke gjennom arrays

En array har en lengde som er lik antall verdier lagret i arrayen:
   arrayen ['a', 'b', 'c', 'd'] har lengde 4.
   arrayen [123456] har lengde 1.
Dette kan vi bruke til å skrive ut alle verdiene i en løkke:
I løkken
   for (var i = plassen første element; i <= plassen til siste element; i++){
     // kode
   }
Vil plassen til første element være 0 og plassen til siste element være lengde-1. Løkken blir dermed
   for (var i = 0; i < arrayen.length; i++){
     // kode
   }

Vi kan for eksempel bruke dette til å skrive ut alle elementene i en array (ikke så spennende):
   var tallArray = [1, 3, 5, 8, 10, 2, 44, 0];
   for(var i = 0; i < tallArray.length; i++){
     console.log(tallArray[i]);
   }

Eller vi kan bruke det til å finne det største elementet i en array (litt mer spennende):
   var storst = tallArray[0];
   for(var i = 0; i < tallArray.length; i++){
     if (tallArray[i] > storst){
       storst = tallArray[i];
     }
   }
   console.log(storst, tallArray.indexOf(storst));

Koden for å finne største verdi i en array kan også utvides til en funksjon som tar inn arrayen og finner det største tallet:
   function finnStorst(tall){
     var storst = tall[0];
     for(var i = 0; i < tall.length; i++){
       if (tall[i] > storst){
         storst = tall[i];
       }
     }
     console.log(storst, tall.indexOf(storst));
   }
   var tallArray = [1, 2, 3, 7, 10, 2];
   finnStorst(tallArray) // evt finnStorst([1, 2, 3, 7, 10, 2])

Løkke gjennom todimensjonale arrays

Todimensjonale arrays er arrays med arrays inne i:
   var A = [
     [1, 2, 3],
     [4, 5, 6]
   ]

For å løkke gjennom disse må vi bruke to for-løkker:
   for(var i = 0; i < A.length; i++){
     for (var j = 0; j < A[i].length; j++){
       console.log(A[i][j]);
     }
   }

Metoden querySelectorAll

Vi husker fra tidligere at metoden querySelector finner første forekomst av et element (querySelector("p")), av et element med en gitt klasse (querySelector(".klassenavn")) eller av et element med en gitt id (querySelector("#idnavn")). Det ble også nevnt at det finnes en metode querySelectorAll som tar inn alle elementer av en gitt type/klasse. Denne har vi spart fordi disse variablene lagres som nettopp en array. For å bearbeide dem må jeg da hente inn alle ved hjelp av en løkke.

Som eksempel kan jeg hente inn alle td-elementer i dette dokumentet og gi dem rosa bakgrunn:
   var tdEl = document.querySelectorAll("td");
   for(var i = 0; i < tdEl.length; i++){
     tdEl[i].style.backgroundColor = "pink";
   }

   < input type="checkbox" class="idrett" value="langrenn">Langrenn

   < input type="checkbox" class="idrett" value="skihopping">Skihopping

Scriptet for dette:
   var idrettEl = document.querySelectorAll(".idrett");
   var skrivUtEl = document.querySelector("#skrivUt");
   var sendEl = document.querySelector("#send");
   sendEl.addEventListener("click", sjekkSkjema);
   function sjekkSkjema(){
     skrivUtEl.innerHTML += 'Du liker å drive med < ul >';
     for(i = 0; i < idrettEl.length; i++){
       if(idrettEl[i].checked){
         skrivUtEl.innerHTML += '< li >' + idrettEl[i].value + '< /li >';
       }
     }
     skrivUtEl.innerHTML += '< /ul >';
   }
